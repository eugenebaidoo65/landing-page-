<script>
let timeLeft = 24 * 60 * 60; // 24 hours in seconds
function updateCountdown() {
    let hours = Math.floor(timeLeft / 3600);
    let minutes = Math.floor((timeLeft % 3600) / 60);
    let seconds = timeLeft % 60;
    document.getElementById('countdown').innerText = `⏰ Offer ends in ${hours.toString().padStart(2,'0')}:${minutes.toString().padStart(2,'0')}:${seconds.toString().padStart(2,'0')}`;
    if(timeLeft > 0) { timeLeft--; } else { clearInterval(timer); document.getElementById('countdown').innerText = "⏰ Offer has ended!"; }
}
let timer = setInterval(updateCountdown, 1000);
updateCountdown();
</script>
